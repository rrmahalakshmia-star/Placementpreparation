import { useState, useRef, useEffect } from "react";

const SYSTEM_PROMPT = `You are a personal Placement Preparation Mentor helping a student get placed in a top IT company.

Rules to follow in every response:
- Explain concepts in simple English and Tanglish (Tamil + English mix) whenever it helps clarity.
- Teach step by step as if the student is a beginner.
- If asked coding questions, first explain the logic, then provide optimized code in C, C++, Java, and Python.
- Focus areas: aptitude, logical reasoning, verbal ability, technical interview questions, HR interview questions, group discussion, resume building, and coding interview preparation.
- For technical subjects: give interview-focused explanations, important points, common mistakes, and expected interview questions.
- Give short revision notes after every topic (use 📝 Short Notes section).
- If the student makes mistakes, correct them politely and explain why.
- Suggest a daily study plan when asked.
- Ask follow-up questions to test understanding after explaining topics.
- Keep answers accurate, practical, and placement-oriented.
- Include real interview tips and company-specific strategies when possible (TCS, Infosys, Wipro, Accenture, Cognizant, Amazon, etc.).
- Motivate the student to stay consistent.
- Use emojis to make responses engaging.
- Format responses with clear sections using markdown-style headers (bold text).
- For code, clearly label each language.`;

const QUICK_TOPICS = [
  { label: "📅 Daily Study Plan", prompt: "Give me a daily study plan for placement preparation" },
  { label: "🧮 Aptitude Tips", prompt: "Teach me aptitude shortcuts for time and work problems" },
  { label: "💻 DSA Roadmap", prompt: "Give me a DSA roadmap for placement" },
  { label: "🗣️ HR Questions", prompt: "What are the top 10 HR interview questions and how to answer them?" },
  { label: "📝 Resume Tips", prompt: "How to build a strong placement resume as a fresher?" },
  { label: "🔢 Arrays Basics", prompt: "Explain arrays for beginners with interview questions and code in C, C++, Java, Python" },
  { label: "🔁 Sorting Algorithms", prompt: "Explain bubble sort and quick sort with code in C, C++, Java, Python" },
  { label: "🏢 TCS Prep", prompt: "How to prepare specifically for TCS placement?" },
];

const PROGRESS_TOPICS = [
  "Aptitude", "Logical Reasoning", "Verbal", "Arrays", "Strings",
  "Linked List", "Trees", "Graphs", "DP", "OS", "DBMS", "Networks", "OOP", "HR Prep"
];

function TypingIndicator() {
  return (
    <div style={{ display: "flex", alignItems: "center", gap: 6, padding: "12px 16px" }}>
      <div style={{ width: 8, height: 8, borderRadius: "50%", background: "#6366f1", animation: "bounce 1.2s infinite" }} />
      <div style={{ width: 8, height: 8, borderRadius: "50%", background: "#6366f1", animation: "bounce 1.2s 0.2s infinite" }} />
      <div style={{ width: 8, height: 8, borderRadius: "50%", background: "#6366f1", animation: "bounce 1.2s 0.4s infinite" }} />
      <span style={{ color: "#94a3b8", fontSize: 13, marginLeft: 4 }}>Mentor is thinking...</span>
    </div>
  );
}

function MessageBubble({ msg }) {
  const isUser = msg.role === "user";
  return (
    <div style={{
      display: "flex",
      justifyContent: isUser ? "flex-end" : "flex-start",
      marginBottom: 16,
      alignItems: "flex-end",
      gap: 8,
    }}>
      {!isUser && (
        <div style={{
          width: 36, height: 36, borderRadius: "50%",
          background: "linear-gradient(135deg, #6366f1, #8b5cf6)",
          display: "flex", alignItems: "center", justifyContent: "center",
          fontSize: 16, flexShrink: 0, boxShadow: "0 2px 8px rgba(99,102,241,0.4)"
        }}>🎓</div>
      )}
      <div style={{
        maxWidth: "78%",
        background: isUser
          ? "linear-gradient(135deg, #6366f1, #8b5cf6)"
          : "rgba(30, 32, 48, 0.95)",
        color: isUser ? "#fff" : "#e2e8f0",
        padding: "12px 16px",
        borderRadius: isUser ? "18px 18px 4px 18px" : "18px 18px 18px 4px",
        fontSize: 14,
        lineHeight: 1.7,
        boxShadow: isUser
          ? "0 4px 15px rgba(99,102,241,0.35)"
          : "0 2px 10px rgba(0,0,0,0.3)",
        border: isUser ? "none" : "1px solid rgba(99,102,241,0.15)",
        whiteSpace: "pre-wrap",
        wordBreak: "break-word",
      }}>
        {msg.content}
      </div>
      {isUser && (
        <div style={{
          width: 36, height: 36, borderRadius: "50%",
          background: "linear-gradient(135deg, #f59e0b, #ef4444)",
          display: "flex", alignItems: "center", justifyContent: "center",
          fontSize: 16, flexShrink: 0,
        }}>🧑‍💻</div>
      )}
    </div>
  );
}

export default function PlacementMentor() {
  const [messages, setMessages] = useState([
    {
      role: "assistant",
      content: `🎓 **Vanakkam! Welcome to your Personal Placement Mentor!**

I'm here to help you crack placements at top IT companies like TCS, Infosys, Wipro, Amazon, and more! 💪

**What I can help you with:**
📊 Aptitude & Logical Reasoning
💻 Coding & DSA (with C, C++, Java, Python)
🗣️ Technical & HR Interviews
📝 Resume Building
🏢 Company-specific Preparation
📅 Daily Study Plans

**Nee ready-ah? Let's start!** 🚀

Pick a topic from the quick buttons below, or type your question directly. Don't hesitate — no question is too basic here!`
    }
  ]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const [progress, setProgress] = useState({});
  const [showProgress, setShowProgress] = useState(false);
  const [streamed, setStreamed] = useState("");
  const bottomRef = useRef(null);
  const inputRef = useRef(null);

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages, loading, streamed]);

  const sendMessage = async (text) => {
    const userMsg = text || input.trim();
    if (!userMsg || loading) return;
    setInput("");
    const newMessages = [...messages, { role: "user", content: userMsg }];
    setMessages(newMessages);
    setLoading(true);
    setStreamed("");

    try {
      const apiMessages = newMessages.map(m => ({ role: m.role, content: m.content }));
      const response = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-6",
          max_tokens: 1000,
          system: SYSTEM_PROMPT,
          messages: apiMessages,
        }),
      });
      const data = await response.json();
      const reply = data.content?.map(b => b.text || "").join("") || "Sorry, I couldn't process that. Please try again!";
      setMessages(prev => [...prev, { role: "assistant", content: reply }]);
    } catch (e) {
      setMessages(prev => [...prev, { role: "assistant", content: "⚠️ Connection error. Please check your internet and try again!" }]);
    }
    setLoading(false);
    setStreamed("");
    inputRef.current?.focus();
  };

  const toggleProgress = (topic) => {
    setProgress(p => ({ ...p, [topic]: p[topic] === "done" ? undefined : "done" }));
  };

  const doneCount = Object.values(progress).filter(v => v === "done").length;

  return (
    <div style={{
      minHeight: "100vh",
      background: "linear-gradient(135deg, #0f0f1a 0%, #1a1a2e 50%, #16213e 100%)",
      fontFamily: "'Inter', 'Segoe UI', sans-serif",
      display: "flex",
      flexDirection: "column",
    }}>
      <style>{`
        @keyframes bounce {
          0%, 80%, 100% { transform: translateY(0); }
          40% { transform: translateY(-8px); }
        }
        @keyframes fadeIn {
          from { opacity: 0; transform: translateY(10px); }
          to { opacity: 1; transform: translateY(0); }
        }
        ::-webkit-scrollbar { width: 6px; }
        ::-webkit-scrollbar-track { background: transparent; }
        ::-webkit-scrollbar-thumb { background: rgba(99,102,241,0.4); border-radius: 3px; }
        textarea { resize: none; outline: none; }
        textarea::placeholder { color: #64748b; }
      `}</style>

      {/* Header */}
      <div style={{
        background: "rgba(15,15,26,0.95)",
        borderBottom: "1px solid rgba(99,102,241,0.2)",
        padding: "14px 20px",
        display: "flex",
        alignItems: "center",
        justifyContent: "space-between",
        backdropFilter: "blur(10px)",
        position: "sticky",
        top: 0,
        zIndex: 100,
      }}>
        <div style={{ display: "flex", alignItems: "center", gap: 12 }}>
          <div style={{
            width: 44, height: 44, borderRadius: "50%",
            background: "linear-gradient(135deg, #6366f1, #8b5cf6)",
            display: "flex", alignItems: "center", justifyContent: "center",
            fontSize: 22, boxShadow: "0 0 20px rgba(99,102,241,0.5)"
          }}>🎓</div>
          <div>
            <div style={{ color: "#fff", fontWeight: 700, fontSize: 17, letterSpacing: 0.3 }}>
              Placement Mentor AI
            </div>
            <div style={{ color: "#6366f1", fontSize: 12, fontWeight: 500 }}>
              🟢 Ready to help you crack placements!
            </div>
          </div>
        </div>
        <button
          onClick={() => setShowProgress(p => !p)}
          style={{
            background: "rgba(99,102,241,0.15)",
            border: "1px solid rgba(99,102,241,0.3)",
            color: "#a5b4fc",
            borderRadius: 10,
            padding: "8px 14px",
            cursor: "pointer",
            fontSize: 13,
            fontWeight: 600,
          }}>
          📊 Progress {doneCount}/{PROGRESS_TOPICS.length}
        </button>
      </div>

      <div style={{ display: "flex", flex: 1, overflow: "hidden" }}>
        {/* Progress Panel */}
        {showProgress && (
          <div style={{
            width: 220,
            background: "rgba(15,15,26,0.98)",
            borderRight: "1px solid rgba(99,102,241,0.15)",
            padding: "16px 12px",
            overflowY: "auto",
            flexShrink: 0,
          }}>
            <div style={{ color: "#a5b4fc", fontWeight: 700, fontSize: 13, marginBottom: 12, letterSpacing: 0.5 }}>
              📋 TOPIC TRACKER
            </div>
            <div style={{
              background: "rgba(99,102,241,0.1)",
              borderRadius: 8, padding: "8px 10px", marginBottom: 14,
              border: "1px solid rgba(99,102,241,0.2)"
            }}>
              <div style={{ color: "#e2e8f0", fontSize: 12, marginBottom: 4 }}>Overall Progress</div>
              <div style={{ background: "rgba(255,255,255,0.1)", borderRadius: 4, height: 6, overflow: "hidden" }}>
                <div style={{
                  width: `${(doneCount / PROGRESS_TOPICS.length) * 100}%`,
                  height: "100%",
                  background: "linear-gradient(90deg, #6366f1, #8b5cf6)",
                  borderRadius: 4,
                  transition: "width 0.4s ease",
                }} />
              </div>
              <div style={{ color: "#94a3b8", fontSize: 11, marginTop: 4 }}>
                {doneCount} of {PROGRESS_TOPICS.length} completed
              </div>
            </div>
            {PROGRESS_TOPICS.map(topic => (
              <div
                key={topic}
                onClick={() => toggleProgress(topic)}
                style={{
                  display: "flex", alignItems: "center", gap: 8,
                  padding: "7px 8px", borderRadius: 8, cursor: "pointer",
                  marginBottom: 4,
                  background: progress[topic] === "done" ? "rgba(99,102,241,0.15)" : "transparent",
                  transition: "background 0.2s",
                }}>
                <div style={{
                  width: 18, height: 18, borderRadius: 5,
                  border: progress[topic] === "done" ? "2px solid #6366f1" : "2px solid rgba(99,102,241,0.3)",
                  background: progress[topic] === "done" ? "#6366f1" : "transparent",
                  display: "flex", alignItems: "center", justifyContent: "center",
                  fontSize: 11, color: "#fff", flexShrink: 0,
                }}>
                  {progress[topic] === "done" ? "✓" : ""}
                </div>
                <span style={{
                  fontSize: 13,
                  color: progress[topic] === "done" ? "#a5b4fc" : "#94a3b8",
                  fontWeight: progress[topic] === "done" ? 600 : 400,
                  textDecoration: progress[topic] === "done" ? "line-through" : "none",
                }}>{topic}</span>
              </div>
            ))}
          </div>
        )}

        {/* Main Chat */}
        <div style={{ flex: 1, display: "flex", flexDirection: "column", overflow: "hidden" }}>
          {/* Quick Topics */}
          <div style={{
            padding: "10px 16px",
            display: "flex",
            gap: 8,
            overflowX: "auto",
            borderBottom: "1px solid rgba(99,102,241,0.1)",
            background: "rgba(15,15,26,0.8)",
            flexShrink: 0,
          }}>
            {QUICK_TOPICS.map(t => (
              <button
                key={t.label}
                onClick={() => sendMessage(t.prompt)}
                disabled={loading}
                style={{
                  background: "rgba(99,102,241,0.12)",
                  border: "1px solid rgba(99,102,241,0.25)",
                  color: "#a5b4fc",
                  borderRadius: 20,
                  padding: "6px 14px",
                  cursor: loading ? "not-allowed" : "pointer",
                  fontSize: 12,
                  fontWeight: 500,
                  whiteSpace: "nowrap",
                  transition: "all 0.2s",
                  flexShrink: 0,
                }}
                onMouseOver={e => e.target.style.background = "rgba(99,102,241,0.25)"}
                onMouseOut={e => e.target.style.background = "rgba(99,102,241,0.12)"}
              >
                {t.label}
              </button>
            ))}
          </div>

          {/* Messages */}
          <div style={{
            flex: 1,
            overflowY: "auto",
            padding: "20px 16px",
          }}>
            {messages.map((msg, i) => (
              <div key={i} style={{ animation: "fadeIn 0.3s ease" }}>
                <MessageBubble msg={msg} />
              </div>
            ))}
            {loading && <TypingIndicator />}
            <div ref={bottomRef} />
          </div>

          {/* Input */}
          <div style={{
            padding: "12px 16px",
            borderTop: "1px solid rgba(99,102,241,0.15)",
            background: "rgba(15,15,26,0.95)",
            display: "flex",
            gap: 10,
            alignItems: "flex-end",
          }}>
            <textarea
              ref={inputRef}
              value={input}
              onChange={e => setInput(e.target.value)}
              onKeyDown={e => {
                if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); sendMessage(); }
              }}
              placeholder="Ask anything about placements... (Press Enter to send, Shift+Enter for new line)"
              rows={1}
              style={{
                flex: 1,
                background: "rgba(30,32,48,0.9)",
                border: "1.5px solid rgba(99,102,241,0.3)",
                borderRadius: 14,
                padding: "12px 16px",
                color: "#e2e8f0",
                fontSize: 14,
                lineHeight: 1.5,
                maxHeight: 120,
                fontFamily: "inherit",
                transition: "border-color 0.2s",
              }}
              onFocus={e => e.target.style.borderColor = "rgba(99,102,241,0.7)"}
              onBlur={e => e.target.style.borderColor = "rgba(99,102,241,0.3)"}
            />
            <button
              onClick={() => sendMessage()}
              disabled={loading || !input.trim()}
              style={{
                background: loading || !input.trim()
                  ? "rgba(99,102,241,0.3)"
                  : "linear-gradient(135deg, #6366f1, #8b5cf6)",
                border: "none",
                borderRadius: 14,
                width: 48,
                height: 48,
                display: "flex",
                alignItems: "center",
                justifyContent: "center",
                cursor: loading || !input.trim() ? "not-allowed" : "pointer",
                fontSize: 20,
                flexShrink: 0,
                boxShadow: loading || !input.trim() ? "none" : "0 4px 15px rgba(99,102,241,0.4)",
                transition: "all 0.2s",
              }}>
              🚀
            </button>
          </div>
          <div style={{ textAlign: "center", color: "#475569", fontSize: 11, padding: "4px 0 8px" }}>
            Your personal AI mentor — available 24/7 for placement prep 💪
          </div>
        </div>
      </div>
    </div>
  );
}
